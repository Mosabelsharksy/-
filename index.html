async function sendQRWhatsApp(i) { let s = students[i]; let canvas = document.querySelector("#qArea canvas"); if (canvas) { try { // 1. تحويل الكانفاس لـ Blob (بيانات ملف حقيقية) const blob = await new Promise(resolve => canvas.toBlob(resolve, 'image/png')); const url = URL.createObjectURL(blob); // 2. إنشاء لينك التحميل وتجربة طريقتين للأمان const downloadLink = document.createElement('a'); downloadLink.href = url; downloadLink.download = `QR-${s.name}.png`; // إضافة اللينك لـ DOM (مهم جداً للموبايل عشان يشتغل) document.body.appendChild(downloadLink); downloadLink.click(); document.body.removeChild(downloadLink); // تنظيف الذاكرة setTimeout(() => URL.revokeObjectURL(url), 100); // 3. تنبيه بسيط (اختياري) alert("✅ تم حفظ الكود في معرض الصور (أو التنزيلات) بنجاح"); } catch (e) { console.error("Download failed", e); alert("حدث خطأ أثناء تحميل الصورة، جرب مرة أخرى"); } } // 4. فتح الواتساب كالمعتاد let msg = `أهلاً يا *${s.name}*.. 🌟 \nده كود الدخول الخاص بك في السنتر. \n\nيرجى إرفاق الصورة التي تم تحميلها الآن وإظهارها عند الدخول. \n\nبالتوفيق يا بطل! ✨`; let phone = s.sPhone.startsWith('0') ? '2' + s.sPhone : s.sPhone; window.open(`https://wa.me/${phone}?text=${encodeURIComponent(msg)}`, '_blank'); }
